        patterns_two = 0
        for i in range(1, n):
            patterns_one, patterns_two = (patterns_one + patterns_two) * (k - 1), patterns_one

再帰 + cache（デコレーター）
```python
class Solution:
    def cache(func):
        memo = {}
        def wrapper(*args):
            if args not in memo:
                memo[args] = func(*args)
            return memo[args]
        return wrapper
    @cache
    def countWays(self, n, k):
        if n == 1:
            return k
        if n == 2:
            return k * k
        return (k - 1) * (self.countWays(n - 1, k) + self.countWays(n - 2, k))
```

## 最終コード

class Solution:
    def countWays(self, n: int, k: int) -> int:
        if n <= 0 or k <= 0:
            raise ValueError("Invalid arguments.")
        last_two_same = k
        last_two_different = 0
        for i in range(1, n):
            last_two_same, last_two_different = (last_two_same + last_two_different) * (k - 1), last_two_same
        return last_two_same + last_two_different
