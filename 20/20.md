class Solution:
    def isValid(self, s: str) -> bool:
        opener_to_closer = {'(': ')', '{': '}', '[': ']'}
        stack = []
        for char in s:
            if char in opener_to_closer.keys():
                stack.append(char)
            else:
                if not stack:
                    return False
                opener = stack.pop()
                if char != opener_to_closer[opener]:
                    return False
        if stack:
            return False
        return True
```

def isValid(s: str) -> bool:
    # 空文字列は有効と見なす
    if not s:
        return True
    
    # 括弧のペアを辞書で定義
    brackets = {
        '(': ')',
        '{': '}',
        '[': ']'
    }
    
    # スタックを使って括弧を管理
    stack = []
    
    # 文字列の各文字をチェック
    for char in s:
        # 開き括弧ならスタックに追加
        if char in brackets:
            stack.append(char)
        # 閉じ括弧なら
        else:
            # スタックが空、または対応する開き括弧がない場合は無効
            if not stack or brackets[stack.pop()] != char:
                return False
    
    # すべての括弧が適切に閉じられていればスタックは空になるはず
    return len(stack) == 0
```

```python
class Solution:
    def isValid(self, s: str) -> bool:
        """与えられた文字列 s が有効なかっこ列か判定する.

        スタックに未消化の開きかっこを積み、
        閉じかっこが来たら直前の開きかっこが
        対応ペアかどうかを調べる。
        """

        # 対応表: 閉じかっこ -> 開きかっこ
        pair = {')': '(', '}': '{', ']': '['}

        stack: list[str] = []

        for ch in s:
            if ch in pair.values():          # 開きかっこなら push
                stack.append(ch)
            elif ch in pair:                 # 閉じかっこなら
                if not stack or stack.pop() != pair[ch]:
                    return False             # 対応しなければ失敗
            else:
                # 問題の制約外の文字が混じっていた場合
                return False

        return not stack                      # すべて消化できていれば成功
```
