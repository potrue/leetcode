```python
class Solution:
    def maxAreaOfIsland(self, grid: List[List[int]]) -> int:
        height = len(grid)
        width = len(grid[0])
        visited_lands = set()
        def enqueue_if_unvisited_land(queue, i, j):
            if not (0 <= i < height and 0 <= j < width):
                return
            if grid[i][j] == 1 and (i, j) not in visited_lands:
                queue.append((i, j))
                visited_lands.add((i, j))
        def check_area_of_island(starting_i, starting_j):
            area = 0
            queue = []
            enqueue_if_unvisited_land(queue, starting_i, starting_j)
            while queue:
                i, j = queue.pop()
                for i_diff, j_diff in [(1, 0), (-1, 0), (0, 1), (0, -1)]:
                    enqueue_if_unvisited_land(queue, i + i_diff, j + j_diff)
                area += 1
            return area
        max_area = 0
        for i in range(height):
            for j in range(width):
                if grid[i][j] == 1 and (i, j) not in visited_lands:
                    max_area = max(max_area, check_area_of_island(i, j))
        return max_area
```

## 最終コード

```python
class Solution:
    def maxAreaOfIsland(self, grid: List[List[int]]) -> int:
        height = len(grid)
        width = len(grid[0])
        visited_lands = set()

        def check_area_of_island(starting_i, starting_j):
            visited_lands.add((starting_i, starting_j))
            frontiers = [(starting_i, starting_j)]
            area = 0
            while frontiers:
                area += 1
                fi, fj = frontiers.pop()
                directions = ((1, 0), (-1, 0), (0, 1), (0, -1))
                for di, dj in directions:
                    i, j = fi + di, fj + dj
                    if (0 <= i < height and 0 <= j < width and
                        grid[i][j] == 1 and (i, j) not in visited_lands):
                        visited_lands.add((i, j))
                        frontiers.append((i, j))
            return area

        max_area = 0
        for i in range(height):
            for j in range(width):
                if grid[i][j] == 1 and (i, j) not in visited_lands:
                    max_area = max(max_area, check_area_of_island(i, j))
        return max_area
```
