```python
class Solution:
    def maxAreaOfIsland(self, grid: List[List[int]]) -> int:
        visited_lands = set()
        num_rows = len(grid)
        num_columns = len(grid[0])
        def is_unvisited_land(r, c):
            return (
                0 <= r < num_rows and
                0 <= c < num_columns and
                grid[r][c] == 1 and
                (r, c) not in visited_lands
            )
        def measure_island(starting_r, starting_c):
            area = 0
            visited_lands.add((starting_r, starting_c))
            frontier = [(starting_r, starting_c)]
            while frontier:
                r, c = frontier.pop()
                directions = [(1, 0), (0, 1), (0, -1), (-1, 0)]
                for dr, dc in directions:
                    new_r = r + dr
                    new_c = c + dc
                    if is_unvisited_land(new_r, new_c):
                        visited_lands.add((new_r, new_c))
                        frontier.append((new_r, new_c))
                area += 1
            return area
        max_area = 0
        for r in range(num_rows):
            for c in range(num_columns):
                if is_unvisited_land(r, c):
                    area = measure_island(r, c)
                    max_area = max(max_area, area)
        return max_area
```
