```python
def num_required_meeting_rooms(intervals: List[List[int]]) -> int:
    events = []
    for start_time, end_time in intervals:
        events.append((start_time, 1))
        events.append((end_time, -1))
    events.sort()
    num_rooms = 0
    max_num_rooms = 0
    for time, num_change in events:
        num_rooms += num_change
        max_num_rooms = max(max_num_rooms, num_rooms)
    return max_num_rooms
```
