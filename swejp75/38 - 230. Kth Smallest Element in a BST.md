### Solution 1

```python
class Solution:
    def kthSmallest(self, root: Optional[TreeNode], k: int) -> int:
        count = 0
        def kthSmallestHelper(node: Optional[TreeNode]):
            nonlocal count
            if not node:
                return
            value = kthSmallestHelper(node.left)
            if value is not None:
                return value
            count += 1
            if count == k:
                return node.val
            value = kthSmallestHelper(node.right)
            if value is not None:
                return value
        return kthSmallestHelper(root)
```
