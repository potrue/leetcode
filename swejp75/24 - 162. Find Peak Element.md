### Solution 1

```python
class Solution:
    def findPeakElement(self, nums: List[int]) -> int:
        left = 0
        right = len(nums) - 1
        while left <= right:
            mid = (left + right) // 2
            greater_than_left = mid - 1 < 0 or nums[mid] > nums[mid - 1]
            greater_than_right = mid + 1 >= len(nums) or nums[mid] > nums[mid + 1]
            if greater_than_left and greater_than_right:
                return mid
            if not greater_than_left:
                right = mid - 1
            else:
                left = mid + 1
```
