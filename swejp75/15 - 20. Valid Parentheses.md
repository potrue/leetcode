```python
class Solution:
    def isValid(self, s: str) -> bool:
        opener_to_closer = {
            '(': ')',
            '{': '}',
            '[': ']',
        }
        stack = []
        for char in s:
            if char in opener_to_closer:
                stack.append(char)
            else:
                if not stack or char != opener_to_closer[stack.pop()]:
                    return False
        if stack:
            return False
        return True
```
