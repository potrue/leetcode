### Solution 1

```python
class Solution:
    def rightSideView(self, root: Optional[TreeNode]) -> List[int]:
        result = []
        nodes = [(root, 0)]
        while nodes:
            node, depth = nodes.pop()
            if not node:
                continue
            if depth >= len(result):
                result.append(node.val)
            nodes.append((node.left, depth + 1))
            nodes.append((node.right, depth + 1))
        return result
```

### Solution 2

BFSでとらえる

```python
class Solution:
    def rightSideView(self, root: Optional[TreeNode]) -> List[int]:
        if not root:
            return []
        result = []
        nodes = [root]
        while nodes:
            result.append(nodes[-1].val)
            new_nodes = []
            for node in nodes:
                if node.left:
                    new_nodes.append(node.left)
                if node.right:
                    new_nodes.append(node.right)
            nodes = new_nodes
        return result
```
