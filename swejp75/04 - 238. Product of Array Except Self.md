```python
class Solution:
    def productExceptSelf(self, nums: List[int]) -> List[int]:
        result = [1] * len(nums)
        cumprod_left = 1
        for i in range(len(nums)):
            result[i] *= cumprod_left
            cumprod_left *= nums[i]
        cumprod_right = 1
        for i in reversed(range(len(nums))):
            result[i] *= cumprod_right
            cumprod_right *= nums[i]
        return result
```
