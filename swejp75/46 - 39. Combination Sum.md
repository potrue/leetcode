```python
class Solution:
    def combinationSum(self, candidates: List[int], target: int) -> List[List[int]]:
        result = []
        def combinationSumHelper(index, combination, combination_sum):
            if combination_sum > target:
                return
            elif combination_sum == target:
                result.append(combination[:])
                return
            if index >= len(candidates):
                return
            combination.append(candidates[index])
            combinationSumHelper(index, combination, combination_sum + candidates[index])
            combination.pop()
            combinationSumHelper(index + 1, combination, combination_sum)
        combinationSumHelper(0, [], 0)
        return result
```
