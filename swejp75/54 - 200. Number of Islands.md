```python
class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        visited_lands = set()
        num_rows = len(grid)
        num_columns = len(grid[0])
        def is_unvisited_land(r, c):
            return (
                0 <= r < num_rows and
                0 <= c < num_columns and
                grid[r][c] == "1" and
                (r, c) not in visited_lands
            )
        def visit_island(starting_r, starting_c):
            visited_lands.add((starting_r, starting_c))
            frontier = [(starting_r, starting_c)]
            while frontier:
                r, c = frontier.pop()
                dirs = [(1, 0), (0, 1), (0, -1), (-1, 0)]
                for dr, dc in dirs:
                    new_r = r + dr
                    new_c = c + dc
                    if is_unvisited_land(new_r, new_c):
                        visited_lands.add((new_r, new_c))
                        frontier.append((new_r, new_c))
        num_islands = 0
        for r in range(num_rows):
            for c in range(num_columns):
                if is_unvisited_land(r, c):
                    visit_island(r, c)
                    num_islands += 1
        return num_islands
```
