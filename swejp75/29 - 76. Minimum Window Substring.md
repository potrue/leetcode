```python
class Solution:
    def minWindow(self, s: str, t: str) -> str:
        if not t:
            return ""
        target = Counter(t)
        counter = defaultdict(int)
        num_letters = len(target)
        num_sufficient_letters = 0
        optimal_pair = None
        left = right = 0
        while right < len(s):
            while right < len(s) and num_sufficient_letters < num_letters:
                c = s[right]
                if c in target:
                    counter[c] += 1
                    if counter[c] == target[c]:
                        num_sufficient_letters += 1
                right += 1
            while left < right and num_sufficient_letters >= num_letters:
                if optimal_pair == None or right - left < optimal_pair[1] - optimal_pair[0]:
                    optimal_pair = (left, right)
                c = s[left]
                if c in target:
                    counter[c] -= 1
                    if counter[c] < target[c]:
                        num_sufficient_letters -= 1
                left += 1
        if not optimal_pair:
            return ""
        return s[optimal_pair[0] : optimal_pair[1]]
```
