```
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        def add_nodes(n1, n2, tail, rolled_up):
            if not n1 and not n2 and not rolled_up:
                return
            val = 0
            if rolled_up:
                val += 1
            if n1:
                val += n1.val
                n1 = n1.next
            if n2:
                val += n2.val
                n2 = n2.next
            new_rolled_up = val >= 10
            if new_rolled_up:
                val -= 10
            new_node = ListNode(val)
            tail.next = new_node
            tail = new_node
            add_nodes(n1, n2, tail, new_rolled_up)
        dummy = ListNode(None)
        tail = dummy
        n1 = l1
        n2 = l2
        add_nodes(n1, n2, tail, False)
        return dummy.next
```

繰り上げを表す変数は適当にrolled_upとつけたが、英語で繰り上げ/繰り下げはcarry/borrowというらしい。

10っていう定数がコード中に二回出てくるのはよくなさそう。

```python
class Solution:
    def addTwoNumbers(self, l1: Optional[ListNode], l2: Optional[ListNode]) -> Optional[ListNode]:
        dummy = ListNode(None)
        tail = dummy
        carrying = 0
        while l1 or l2 or carrying:
            total = carrying
            if l1:
                total += l1.val
                l1 = l1.next
            if l2:
                total += l2.val
                l2 = l2.next
            carrying, val = divmod(total, 10)
            tail.next = ListNode(val)
            tail = tail.next
        return dummy.next
```
