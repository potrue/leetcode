class Solution:
    def ladderLength(self, beginWord: str, endWord: str, wordList: List[str]) -> int:
        def is_adjacent_word(word1, word2):
            distance = 0
            for i in range(len(beginWord)):
                if word1[i] != word2[i]:
                    distance += 1
            return distance == 1

        if endWord not in wordList:
            return 0
        wordList.append(beginWord)
        adjacent_words = {word: set() for word in wordList}
        for word1, word2 in combinations(wordList, 2):
            if is_adjacent_word(word1, word2):
                adjacent_words[word1].add(word2)
                adjacent_words[word2].add(word1)

        visited_words = set([beginWord])
        queue = deque([(beginWord, 1)])
        while queue:
            word, depth = queue.popleft()
            for adjacent_word in adjacent_words[word]:
                if adjacent_word == endWord:
                    return depth + 1
                if adjacent_word not in visited_words:
                    queue.append((adjacent_word, depth + 1))
                    visited_words.add(adjacent_word)
        return 0

## 最終コード

```python
class Solution:
    def ladderLength(self, beginWord: str, endWord: str, wordList: List[str]) -> int:
        def patterns(word):
            return [word[:i] + "*" + word[i + 1:] for i in range(len(word))]
        pattern_to_words = defaultdict(list)
        wordList.append(beginWord)
        for word in wordList:
            for pattern in patterns(word):
                pattern_to_words[pattern].append(word)
        visited_words = set([beginWord])
        queue = deque([(beginWord, 1)])
        while queue:
            word, length = queue.popleft()
            for pattern in patterns(word):
                for adjacent_word in pattern_to_words[pattern]:
                    new_length = length + 1
                    if adjacent_word == endWord:
                        return new_length
                    if adjacent_word in visited_words:
                        continue
                    visited_words.add(adjacent_word)
                    queue.append((adjacent_word, new_length))
        return 0
```
