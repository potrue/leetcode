## 何も見ずに解いてみる

```cpp
class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {
        int start = 0;
        int sum = 0;
        int min_length = INT_MAX;
        for (int end = 0; end < nums.size(); ++end) {
            sum += nums[end];
            while (sum - nums[start] >= target) {
                sum -= nums[start];
                ++start;
            }
            if (sum >= target) {
                min_length = min(min_length, end - start + 1);
            }
        }
        if (min_length == INT_MAX) {
            return 0;
        }
        return min_length;
    }
};
```

class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {
        int start = 0;
        int end = 0;
        int sum = 0;
        int min_length = std::numeric_limits<int>::max();
        while (end < nums.size()) {
            while (sum < target && end < nums.size()) {
                sum += nums[end++];
            }
            while (sum >= target) {
                min_length = min(min_length, end - start);
                sum -= nums[start++];
            }
        }
        if (min_length == std::numeric_limits<int>::max()) {
            return 0;
        }
        return min_length;
    }
};

## 最終コード

```cpp
class Solution {
public:
    int minSubArrayLen(int target, vector<int>& nums) {
        int min_length = std::numeric_limits<int>::max();
        int start = 0;
        int sum = 0;
        for (int end = 0; end < nums.size(); ++end) {
            sum += nums[end];
            while (sum >= target) {
                min_length = min(min_length, end - start + 1);
                sum -= nums[start];
                ++start;
            }
        }
        if (min_length == std::numeric_limits<int>::max()) {
            return 0;
        }
        return min_length;
    }
};
```
