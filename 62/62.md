## 何も見ずに解いてみる

```cpp
class Solution {
public:
    int uniquePaths(int m, int n){
        vector<vector<int>> num_paths(n, vector<int>(m));
        num_paths[0][0] = 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (i != 0) {
                    num_paths[i][j] += num_paths[i - 1][j];
                }
                if (j != 0) {
                    num_paths[i][j] += num_paths[i][j - 1];
                }
            }
        }
        return num_paths[n - 1][m - 1];
    }
};
```

combinationの数を計算して返す方法もある

```cpp
class Solution {
public:
    int uniquePaths(int m, int n) {
        return combination(m + n - 2, m - 1);
    }
private:
    int combination(int n, int r) {
        if (r > n - r) {
            r = n - r;
        }
        long long result = 1;
        for (int i = 0; i < r; i++) {
            result = result * (n - i) / (1 + i);
        }
        return result;
    }
};
```

## 最終コード
```cpp
class Solution {
public:
    int uniquePaths(int m, int n) {
        vector<vector<int>> unique_paths(n, vector<int>(m, 0));
        unique_paths[0][0] = 1;
        for (int i = 0; i < n; i++) {
            for (int j = 0; j < m; j++) {
                if (i != 0) {
                    unique_paths[i][j] += unique_paths[i - 1][j];
                }
                if (j != 0) {
                    unique_paths[i][j] += unique_paths[i][j - 1];
                }
            }
        }
        return unique_paths.back().back();
    }
};
```
