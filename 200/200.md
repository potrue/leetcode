class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        height = len(grid)
        width = len(grid[0])
        visited_lands = set()
        def enqueue_if_unvisited_land(queue, i, j):

                0 <= i < height and 0 <= j < width and
                grid[i][j] == "1" and (i, j) not in visited_lands
            ):
                queue.append((i, j))
                visited_lands.add((i, j))
        def visit_island(starting_i, starting_j):
            frontiers = [(starting_i, starting_j)]
            while frontiers:
                i, j = frontiers.pop()
                for di, dj in ((0, 1), (0, -1), (1, 0), (-1, 0)):
                    new_i, new_j = i + di, j + dj
                    enqueue_if_unvisited_land(frontiers, new_i, new_j)
        num_islands = 0
        for i in range(height):
            for j in range(width):
                if grid[i][j] == "1" and (i, j) not in visited_lands:
                    visit_island(i, j)
                    num_islands += 1
        return num_islands

## 最終コード

```python
class Solution:
    def numIslands(self, grid: List[List[str]]) -> int:
        num_rows = len(grid)
        num_columns = len(grid[0])
        visited_lands = set()
        def enqueue_if_unvisited_land(queue, r, c):
            if (
                0 <= r < num_rows and 
                0 <= c < num_columns and
                grid[r][c] == "1" and 
                (r, c) not in visited_lands
            ):
                queue.append((r, c))
                visited_lands.add((r, c))
        def visit_island(starting_r, starting_c):
            frontiers = [(starting_r, starting_c)]
            visited_lands.add((starting_r, starting_c))
            while frontiers:
                r, c = frontiers.pop()
                dirs = ((0, 1), (0, -1), (1, 0), (-1, 0))
                for dr, dc in dirs:
                    new_r, new_c = r + dr, c + dc
                    enqueue_if_unvisited_land(frontiers, new_r, new_c)
        num_islands = 0
        for r in range(num_rows):
            for c in range(num_columns):
                if grid[r][c] == "1" and (r, c) not in visited_lands:
                    visit_island(r, c)
                    num_islands += 1
        return num_islands
```

r, cとi, jはどっちが見やすいでしょうか。最初に定義するときにheightとwidthと言ったならyとx、num_rowsやnum_columnsと言ったならrとc、dim1やdim2と言ったならiとjを使うのがよいかもしれません。
